\begin{definition}
  Let
    $R$ be a commutative ring with unity,
    $V$ be a module over $R$.
    $S \subset V$.
  We say that $S$ \textbf{spans} $V$ if
  for any $v \in V$
  there exist $n \in \N$, $\lambda_1, ..., \lambda_n \in R$,
  $v_1, ..., v_n \in V$ such that
  \begin{equation}
    v = \lambda_1 v_1 + ... + \lambda_n v_n.
  \end{equation}
\end{definition}
