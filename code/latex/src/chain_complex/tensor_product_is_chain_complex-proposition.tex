\begin{proposition}
  Let $(A_\bullet, \partial^A)$ and $(B_\bullet, \partial^B)$
  be chain complexes,
  \begin{equation}
    (C_\bullet, \partial^C)
    = (A_\bullet, \partial^A) \otimes (B_\bullet, \partial^B)
  \end{equation}
  be their tensor product (\Cref{idec/chain_complex/tensor_product-definition}).
  Then $(C_\bullet, \partial^C)$ is a chain complex.
\end{proposition}
