\begin{definition}
  Let $m, n \in \N$, $A$ be a real $m \times n$ matrix.
  An $n \times m$ matrix $B$ is called \textbf{Moore-Penrose inverse} or
  \textbf{pseudo-inverse} if
  \begin{subequations}
    \begin{align}
      A B A & = A, \\
      B A B & = B, \\
      (A B)^T & = A B, \\
      (B A)^T & = B A.
    \end{align}
  \end{subequations}
\end{definition}
