\begin{example}
  Consider
    $V = \R^2$ with the standard basis $e = (e_0, e_1)$,
    inner product $g$ making $e$ orthonormal,
    volume $2$-vector $\vol := e_0 \wedge e_1$.
  Then
  \begin{subequations}
    \begin{align}
      & \star_0 1 = e_0 \wedge e_1 = \vol, \\
      & \star_1 e_0 = e_1, \\
      & \star_1 e_1 = - e_0, \\
      & \star_2 e_0 \wedge e_1 = \star_2 \vol = 1.
    \end{align}
  \end{subequations}
\end{example}
