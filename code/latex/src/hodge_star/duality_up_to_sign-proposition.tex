\begin{proposition}
  Let
    $d \in \N$
    $(V, g)$ be an oriented real inner product space of dimension $d$,
    $\vol$ be the volume $d$-vector on $\Lambda^d V$,
    $p \in \N$,
    $\star$ be the Hodge star operator on $\Lambda^\bullet V$.
  Then
  \begin{equation}
    \star_{d - p} \circ \star_p = (-1)^{p (d - p)} \id_{\Lambda^p V}.
  \end{equation}
\end{proposition}
