\begin{proposition}
  Let $R$ be a commutative ring with unity, $n$ be  positive integer.
  Consider the set $R^n$ as a ring with pointwise addition and multiplication,
  and as an $R$-algebra with pointwise scalar multiplication.
  Then
  \begin{equation}
    {\rm Der}_R(R^n) = 0.
  \end{equation}
\end{proposition}
