\begin{corollary}
  Let $R$ be a commutative ring with unity,
  $A$ be a unital, commutative, and associative $R$-algebra.
  Then ${\rm Der}_R(A)$ is a Lie algebra with Lie bracket given by
  \begin{equation}
    [X, Y] := X \circ Y - Y \circ X,\ X, Y \in {\rm Der}_R(A). 
  \end{equation}
\end{corollary}
