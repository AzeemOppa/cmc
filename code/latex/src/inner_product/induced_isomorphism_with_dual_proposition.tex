\begin{proposition}
  Let $(V, g)$ be a finite-dimensional real inner product space.
  Define the map $\tilde{g} \colon V \to V^*$ as follows: for any $v \in V$,
  \begin{equation}
    \tilde{g}(v) := (w \in V \mapsto g(v, w)).
  \end{equation}
  Then $\tilde{g}$ is an isomorphism.
\end{proposition}
