\begin{example}
  Consider the \hyperref[cmc:jagged:definition]{jagged array} of order $3$
  \begin{equation}
    x = (((0, 1, 2), (3, 4)), (), (()), ((5, 6), ())).
  \end{equation}
  Then $x$ has the following representation:
  \begin{equation}
    \begin{split}
      & x.a0 = 4 \\
      & x.a1 = (2, 0, 1, 2) \\
      & x.a2 = (3, 2, 0, 2, 0) \\
      & x.a3 = (0, 1, 2, 3, 4 , 5, 6)
    \end{split}
  \end{equation}
\end{example}
