\begin{proposition}
  Let $R$ be a commutative ring with unity, $S$ be a subset of $R$.
  Then $S$ is a subring with unity if contains $1$ and is closed under
  negation, addition, and multiplication, i.e.,
  for any $a, b \in R$,
  \begin{equation}
    1 \in \R,\ -a \in \R,\ a + b \in \R,\ a b \in R.
  \end{equation}
\end{proposition}
