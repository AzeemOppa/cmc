\begin{proposition}
  $R$ be a commutative ring with unity, $V$ be a Lie algebra over $R$.
  Then the adjoint map is a homorphism of Lie algebras, i.e.,
  for any $x, y \in V$,
  \begin{equation}
    {\rm adj}_{[x, y]}
    = [{\rm adj}_x, {\rm adj}_y]
    := {\rm adj}_x \circ {\rm adj}_y - {\rm adj}_y \circ {\rm adj}_x.
  \end{equation}
\end{proposition}
