\begin{definition}
  Let
    $R$ be a commutative ring with unity,
    $(V, [\cdot, \cdot]_V)$ and
    $(W, [\cdot, \cdot]_W)$ be Lie algebras over $R$,
    $\varphi \in \Hom_R(V, W)$.
  We say that $\varphi$ is a \textbf{Lie algebra homomorphism}
  if for any $u, v \in V$,
  \begin{equation}
    \varphi([u, v]_V) = [\varphi u, \varphi v]_W.
  \end{equation}
\end{definition}
