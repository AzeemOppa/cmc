\begin{proposition}
  Let
    $R$ be a commutative ring with unity,
    $D \in \N$,
    $V$ be a free module over $R$ of dimension $D$,
    $v_0, ..., v_{D - 1} \in V$.
  Then $(v_0, ..., v_{D - 1})$ form a basis of $V$ if and only if
  \begin{equation}
    v_0 \wedge ... \wedge v_{D - 1}
  \end{equation}
  forms a basis of the $1$-dimensional module $\Lambda^D V$.
\end{proposition}
