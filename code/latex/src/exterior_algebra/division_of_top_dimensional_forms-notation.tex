\begin{notation}
  Let
    $V$ be an $1$-dimensional real vector space,
    $v \in V$,
    $w \in V \setminus \{0\}$.
  By $v / w$ denote the unique $\lambda \in \R$ such that $v = \lambda w$.
  (The uniqueness follows from the fact that the nonzero vector $w$ forms a
  basis of $V$.)
\end{notation}
