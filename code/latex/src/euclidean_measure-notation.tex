\begin{notation}
  Let $d \in \N$,
  $P$ be a polytope of dimension $d$.
  The \textbf{(Euclidean) measure} of $P$ is denoted by
  \begin{equation}
    \mu_d(P) (= \mu(P)).
  \end{equation}
  If $P$ is with standard physical dimensions
  (e.g., it has not been non-dimensionalised),
  then $\mu_d(P)$ is of physical dimension $[L^d]$.
\end{notation}
