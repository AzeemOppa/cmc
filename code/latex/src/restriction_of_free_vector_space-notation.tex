\begin{notation}
  Let
    $S$ be a set,
    $T$ be a subset of $S$,
    $V$ be a real vector space,
    $u \in \Hom_\R({\rm Free}_\R(S), V)$.
  We will denote by
  \begin{equation}
    \restrict{u}{T} \in \Hom_\R({\rm Free}_\R(T), V)
  \end{equation}
  the map defined in the same way of $u$ but acting on formal linear
  combinations of the elements in $T$.
\end{notation}
