\begin{proposition}
  Let
    $d \in \N$,
    $M$ be a \hyperref[cmc:mesh:definition]{mesh} of dimension $d$,
    $R$ be a commutative ring with unity,
    $\partial$ be a boundary operator on $M$,
    $\delta$ be the corresponding coboundary operator.
  Then
  \begin{equation}
    (C^\bullet(M; R), \delta)
  \end{equation}
  is a cochain complex.
\end{proposition}
