\begin{proposition}
  Let
    $d \in \N$,
    $M$ be a \hyperref[idec:mesh:definition]{mesh} of dimension $d$,
    $\partial$ be a boundary operator on $M$m
    $D$ be the discrete differential on $M$.
  Then $(\Omega^\bullet M, D)$ is a cochain complex, i.e.,
  \begin{equation}
    D^2 = 0.
  \end{equation}
\end{proposition}
