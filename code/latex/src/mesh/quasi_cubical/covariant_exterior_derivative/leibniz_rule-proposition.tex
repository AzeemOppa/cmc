\begin{proposition}
  Let
    $K$ be a quasi-cubical mesh,
    $V$ be a vector space,
    $v \in V$,
    $p \in \N$,
    $\sigma^p \in C^p K$.
  Then
  \begin{equation}
    \nabla_p(v \usmile \sigma^p)
    = \nabla_0 v \usmile \sigma^p + v \usmile \delta_p \sigma^p.
  \end{equation}
\end{proposition}
