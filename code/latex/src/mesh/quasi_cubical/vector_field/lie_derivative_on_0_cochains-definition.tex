\begin{definition}
  Let $\mathcal{K}$ be a quasi-cubical mesh.
  The \textbf{Lie derivative on $0$-cochains} is defined as
  \begin{equation}
    L \colon \mathfrak{X} K \to \Hom(C^0 \mathcal{K}, C^0 \mathcal{K}),
    L_X := i_X \circ \delta,\ X \in \mathfrak{X} K.
  \end{equation}
\end{definition}
