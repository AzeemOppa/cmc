\begin{corollary}
  Let
    $d \in \N$,
    $h \in \R^+$
    $K$ a cubical grid of dimension $d$ with size $h$
      with its standard orientation,
    $p \in \{0, ..., d\}$,
    $\pi^p \in C^p K$,
    $b_{p - 1} \in K_{p - 1}$ be an internal cell.
  Then
  \begin{equation}
    (\delta^\star_p \pi^p) b_{p - 1} =
    \frac{1}{h^2}
    \sum_{b_{p - 1} \prec a_p}
      \varepsilon(a_p, b_{p - 1})
      \pi^p(a_p).
  \end{equation}
\end{corollary}
