\begin{proposition}
  Let
    $d \in \N$,
    $M$ be a \hyperref[cmc:mesh:definition]{mesh} of dimension $d$,
    $R$ be a commutative ring with unity,
    $\varepsilon$ be a relative orientation on $M$.
  Then the algebra $(C_\bullet M, \partial)$ is a chain complex, i.e.,
  \begin{equation}
    \partial^2 = 0.
  \end{equation}
\end{proposition}
